## vue节流函数



防止请求请求过多，一般用于边输入边搜索功能
``` js
<script>
    // 节流函数
    const delay = (()=> {
        let timer = 0
        return (callback, ms) => {
            clearTimeout(timer)
            timer = setTimeout(callback, ms)
        }
    })()

    export default {
        data(){
            return{

            }
        },
        // 在方法中使用
        methods:{
            search(){
                delay(() => {
                    // 逻辑内容
                },300)
            }
        }
    }
</script>
```
